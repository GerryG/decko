def load_layouts
  hash = {}
  mod_dirs.each(:layout) do |dirname|
    Dir.foreach(dirname) do |filename|
      next if filename =~ /^\./
      layout_name = filename.gsub(/\.html$/, "")
      hash[layout_name] = File.read File.join(dirname, filename)
    end
  end
  hash
end
