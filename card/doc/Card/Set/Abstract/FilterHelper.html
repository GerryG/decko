def filter_hash
  @filter_hash ||= begin
    filter = Env.params[:filter]
    filter = filter.to_unsafe_h if filter&.respond_to?(:to_unsafe_h)
    filter.is_a?(Hash) ? filter : {}
  end
end

def filter_keys_with_values

    values = filter_param(key)
    values.present? ? [key, values] : next
  end.compact
end
