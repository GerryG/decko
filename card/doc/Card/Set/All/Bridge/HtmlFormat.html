def bridge_link_opts opts={}
  opts[:data-slot-selector"] = bridge_slot_selector
  opts[:remote] = true
  add_class opts, "slotter"
  opts.bury :path, :layout, :overlay
  opts[:path][:view] ||= :content
  opts
end
