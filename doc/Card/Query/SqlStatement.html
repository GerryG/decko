def to_s
  select = fields.reject(&:blank?) * ', '
  where = conditions.reject(&:blank?) * ' and '

  ['(SELECT', distinct, select, 'FROM', tables, joins, 'WHERE', where, group, order, limit, offset, ')'].compact * ' '
end
