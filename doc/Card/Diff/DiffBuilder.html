def initialize(old_version, new_version, opts={})
  @new_version = new_version
  @old_version = old_version
  @lcs_opts = lcs_opts_for_format opts[:format]
  @lcs_opts[:summary] = opts[:summary]
  @dels_cnt = 0
  @adds_cnt = 0
  
  if not @new_version
    @complete = ''
    @summary  = ''
  else
    lcs_diff
  end
end

def lcs_opts_for_format format
  opts = {}
  case format
  when :html
    opts[:exclude] = /^</
  when :text
    opts[:reject] =  /^</
    opts[:postprocess] = Proc.new do |word|
      word.gsub("\n",'<br>')
    end
  when :pointer
    opts[:preprocess] = Proc.new do |word|
      word.gsub('[[','').gsub(']]','')
    end
  else #:raw
    opts[:preprocess] = Proc.new do |word|
      CGI::escapeHTML(word)
    end
  end
  opts
end
