def declare_spec(parent, spec)
  me = self
  example_name = spec["name"]
  @spec_ids << spec["id"]
  backtrace = @example_locations[parent.description + " " + example_name]
  parent.it example_name, {} do
    me.report_spec(spec["id"])
  end
end
